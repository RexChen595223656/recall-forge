```python
def create_material(page, title, content_file):
    """上传材料并返回 material_id"""
    with open(f"tests/test_data/{content_file}") as f:
        content = f.read()
    page.locator("text=添加材料").first.click()
    page.wait_for_timeout(300)
    page.locator('input[placeholder="材料标题"]').fill(title)
    page.locator("textarea").fill(content)
    page.locator("text=下一步").click()
    page.wait_for_timeout(300)
    page.locator("text=确认上传").click()
    page.wait_for_timeout(2000)
    # 从 URL 提取 material_id
    page.wait_for_timeout(1000)
    # 点击材料卡片
    page.locator(".cursor-pointer").first.click()
    page.wait_for_timeout(1000)
    current_url = page.url
    import re
    match = re.search(r'material=(\d+)', current_url)
    return int(match.group(1)) if match else None

def wait_for_questions_ready(page, material_id, timeout=90):
    """轮询直到题目生成完成"""
    import time, requests
    start = time.time()
    while time.time() - start < timeout:
        resp = requests.get(f"http://localhost:8000/api/materials/{material_id}/status")
        data = resp.json()
        if data["status"] == "ready":
            return data["question_count"]
        if data["status"] == "error":
            raise Exception(f"Generation failed: {data.get('error_message')}")
        time.sleep(2)
    raise TimeoutError("Generation timed out")

def answer_question(page, option_index=0):
    """选择第 N 个选项并处理反馈/下一题"""
    options = page.locator("button:has-text('A.')")
    if options.count() == 0:
        # 尝试找答题中的选项按钮
        buttons = page.locator("button")
        option_buttons = []
        for i in range(buttons.count()):
            text = buttons.nth(i).text_content() or ""
            if text.strip() and text[0] in "ABCD":
                option_buttons.append(buttons.nth(i))
        if option_buttons:
            option_buttons[min(option_index, len(option_buttons)-1)].click()
    else:
        options.first.click()
    page.wait_for_timeout(500)

def go_next_or_finish(page):
    """点击下一题或查看结果"""
    for text in ["下一题 →", "查看结果", "返回"]:
        btn = page.locator(f"text={text}")
        if btn.count() > 0:
            btn.click()
            page.wait_for_timeout(300)
            return text
    return None

def complete_quiz_round(page, question_count):
    """完成一轮答题（全部选 A）"""
    for i in range(question_count):
        page.wait_for_selector("text=A.", timeout=10000)
        page.wait_for_timeout(200)
        page.locator("button:has-text('A.')").first.click()
        page.wait_for_timeout(500)
        action = go_next_or_finish(page)
        if action == "查看结果" or action == "返回":
            break

def assert_no_errors(page, errors):
    """检查页面无 JS 错误"""
    page.wait_for_timeout(500)
    assert len(errors) == 0, f"Page errors: {errors}"
```

```python
def test_mastered_after_3_consecutive_correct(page):
    """模拟连续 3 次正确，验证攻克"""
    page, errors = page
    # 准备：创建材料并答题产生错题
    material_id = create_material(page, "攻克测试", "short.txt")
    count = wait_for_questions_ready(page, material_id)
    page.locator(".cursor-pointer").first.click()
    page.wait_for_timeout(1000)
    
    # 第 1 轮：故意全选错，产生错题
    for i in range(count):
        page.wait_for_timeout(300)
        # 选第一个选项（假设不全对）
        btns = page.locator("button")
        option_btns = []
        for j in range(btns.count()):
            t = btns.nth(j).text_content() or ""
            if t.strip() and t[0] in "ABCD" and len(t) > 2:
                option_btns.append(btns.nth(j))
        if option_btns:
            option_btns[0].click()
        page.wait_for_timeout(500)
        go_next_or_finish(page)
    
    page.locator("text=返回").click()
    page.wait_for_timeout(1000)
    
    # 进入错题重做，连续 3 次选对同一道题
    page.locator("text=错题重做").click()
    page.wait_for_timeout(1000)
    
    # 使用 API 直接操作来模拟连续 3 次正确（E2E 中难以精确控制选项正确性）
    import requests
    API = "http://localhost:8000/api"
    # 获取错题列表
    wrong = requests.get(f"{API}/review/wrong-questions?material_id={material_id}").json()
    if wrong:
        qid = wrong[0]["id"]
        correct_ans = wrong[0]["answer"]
        # 连续 3 次提交正确答案
        for _ in range(3):
            resp = requests.post(f"{API}/quiz/submit", json={"question_id": qid, "user_answer": correct_ans}).json()
        # 第 3 次后 mastered=true
        assert resp.get("mastered") == True, f"Expected mastered=true after 3 correct, got {resp}"
        assert resp.get("challenge_streak") == 3
    
    assert_no_errors(page, errors)
```

```python
def test_delete_material_during_generation(page):
    """RACE-01：出题进行中删除材料 → 后台线程不崩溃"""
    page, errors = page
    import requests
    API = "http://localhost:8000/api"
    
    # 创建材料，触发出题
    resp = requests.post(f"{API}/materials", data={
        "title": "竞态删除测试",
        "content": "AI产品经理的核心能力包括理解大模型能力边界、设计RAG架构、工程化Prompt系统、建立多层评估体系和构建数据飞轮五项。" * 3
    })
    mid = resp.json()["id"]
    requests.post(f"{API}/materials/{mid}/generate")
    
    # 立即删除材料（不等出题完成）
    import time
    time.sleep(1)
    resp = requests.delete(f"{API}/materials/{mid}")
    assert resp.status_code == 200
    
    # 验证：不崩溃（后端进程仍在运行）
    health = requests.get(f"{API}/health")
    assert health.status_code == 200
    
    assert_no_errors(page, errors)
```
