### TC-STATS-01：ASC 排序 + 轮次聚类

```python
def test_attempts_sorted_asc_and_session_clustering(db_session):
    from models.database import Attempt, Question
    from datetime import datetime, timedelta
    
    # 创建题目
    q = Question(material_id=1, chunk_id="c1", question="Q", options=["A. a","B. b","C. c","D. d"], answer="A")
    db_session.add(q)
    db_session.commit()
    
    now = datetime.utcnow()
    # 模拟 2 轮答题：第 1 轮 (t+0, t+2min)，第 2 轮 (t+10min, t+11min)
    attempts = [
        Attempt(question_id=q.id, user_answer="A", is_correct=1, created_at=now),
        Attempt(question_id=q.id, user_answer="B", is_correct=0, created_at=now + timedelta(minutes=2)),
        Attempt(question_id=q.id, user_answer="C", is_correct=0, created_at=now + timedelta(minutes=10)),
        Attempt(question_id=q.id, user_answer="A", is_correct=1, created_at=now + timedelta(minutes=11)),
    ]
    for a in attempts:
        db_session.add(a)
    db_session.commit()
    
    # 按 ASC 排序，间隔 >5min 为新轮次
    all_attempts = db_session.query(Attempt).filter(
        Attempt.question_id == q.id
    ).order_by(Attempt.created_at.asc()).all()
    
    sessions = []
    for a in all_attempts:
        if not sessions or (a.created_at - sessions[-1]["start"]).total_seconds() > 300:
            sessions.append({"start": a.created_at, "attempts": []})
        sessions[-1]["attempts"].append(a)
    
    assert len(sessions) == 2, f"Expected 2 sessions, got {len(sessions)}"
    assert len(sessions[0]["attempts"]) == 2  # 第 1 轮 2 题
    assert len(sessions[1]["attempts"]) == 2  # 第 2 轮 2 题
```

## 六、正确率计算 (`test_stats.py`)

### TC-ACC-01：近期 < 10 次用最近 3 轮

```python
def test_recent_accuracy_uses_sessions_when_few_attempts(db_session):
    from models.database import Attempt, Question
    from datetime import datetime, timedelta
    
    q = Question(material_id=1, chunk_id="c1", question="Q", options=["A. a","B. b","C. c","D. d"], answer="A")
    db_session.add(q)
    db_session.commit()
    
    now = datetime.utcnow()
    # 3 轮：第 1 轮(1/2), 第 2 轮(2/2), 第 3 轮(1/2) → 总共 6 次 (< 10)
    data = [
        # 第 1 轮
        (now - timedelta(days=3), 1), (now - timedelta(days=3) + timedelta(minutes=1), 0),
        # 第 2 轮  
        (now - timedelta(days=1), 1), (now - timedelta(days=1) + timedelta(minutes=1), 1),
        # 第 3 轮
        (now, 1), (now + timedelta(minutes=1), 0),
    ]
    for ts, correct in data:
        db_session.add(Attempt(question_id=q.id, user_answer="A" if correct else "B", is_correct=correct, created_at=ts))
    db_session.commit()
    
    # 提取最近 3 轮 → 第 1、2、3 轮
    all_attempts = db_session.query(Attempt).filter(
        Attempt.question_id == q.id
    ).order_by(Attempt.created_at.asc()).all()
    
    sessions = []
    for a in all_attempts:
        if not sessions or (a.created_at - sessions[-1]["start"]).total_seconds() > 300:
            sessions.append({"start": a.created_at, "correct": 0, "total": 0})
        sessions[-1]["total"] += 1
        if a.is_correct:
            sessions[-1]["correct"] += 1
    
    recent_3 = sessions[-3:]
    total = sum(s["total"] for s in recent_3)
    correct = sum(s["correct"] for s in recent_3)
    recent_acc = round(correct / total * 100, 1) if total > 0 else 0
    
    assert total == 6
    assert recent_acc == round((1+2+1) / 6 * 100, 1)  # 4/6 = 66.7%
```

### TC-ACC-02：近期 ≥ 10 次用 30 天窗口

```python
def test_recent_accuracy_uses_30days_when_enough(db_session):
    from models.database import Attempt, Question
    from datetime import datetime, timedelta
    
    q = Question(material_id=1, chunk_id="c1", question="Q", options=["A. a","B. b","C. c","D. d"], answer="A")
    db_session.add(q)
    db_session.commit()
    
    now = datetime.utcnow()
    # 15 次答题（≥ 10）：12 次在 30 天内(10对2错)，3 次在 35 天前(1对2错)
    for i in range(12):
        ts = now - timedelta(days=i % 20)  # 分散在 20 天内
        db_session.add(Attempt(question_id=q.id, user_answer="A", is_correct=1 if i < 10 else 0, created_at=ts))
    for i in range(3):
        ts = now - timedelta(days=35)  # 35 天前
        db_session.add(Attempt(question_id=q.id, user_answer="B", is_correct=1 if i == 0 else 0, created_at=ts))
    db_session.commit()
    
    cutoff = now - timedelta(days=30)
    all_attempts = db_session.query(Attempt).filter(Attempt.question_id == q.id).all()
    recent = [a for a in all_attempts if a.created_at >= cutoff]
    old = [a for a in all_attempts if a.created_at < cutoff]
    
    assert len(all_attempts) == 15
    assert len(recent) == 12  # 30 天内
    assert len(old) == 3      # 30 天外
    
    recent_acc = round(sum(1 for a in recent if a.is_correct) / len(recent) * 100, 1)
    assert recent_acc == round(10 / 12 * 100, 1)  # 83.3%
```

def test_wrong_questions_filters_correctly(db_session):
